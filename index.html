      from odoo import models, fields, api

class SaleOrder(models.Model):
    _inherit = "sale.order"

    marge = fields.Float(
        compute="_compute_marge",
        store=True,
    )

    @api.depends("order_line.margin")
    def _compute_marge(self):
        for order in self:
            order.marge = sum(
                order.order_line.mapped("margin")
            )
